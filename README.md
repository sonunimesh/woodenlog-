## Installation
```shell
npm install woodenlog --save
```
## Usage
```js
var woodenlog = require('woodenlog');
// woodenlog.configurate(null, 'white', 'green', 'red');
woodenlog.log('just log!');
woodenlog.warn('this is a warning!');
woodenlog.error('this is an error!');
```
